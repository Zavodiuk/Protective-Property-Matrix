## Purpose
Programmatic implementation of the viability calculation mechanism based on the Architectural Blocks.

## Code
```python
from enum import Enum
from typing import List

class State(Enum):
    FALSE = 0       # Нет
    TRUE = 1        # Да
    UNKNOWN = 0.5   # Может быть

class ArchitecturalNode:
    def __init__(self, name: str, state: State = State.UNKNOWN):
        self.name = name
        self.state = state
        self.connections: List['ArchitecturalNode'] = []

    def connect(self, other: 'ArchitecturalNode'):
        self.connections.append(other)

    def __repr__(self):
        return f"Node({self.name}, State={self.state.name})"

class ViabilityEngine:
    """Движок для расчета жизнеспособности (Блок B)"""
    def __init__(self, nodes: List[ArchitecturalNode]):
        self.nodes = nodes

    def calculate_system_viability(self) -> float:
        """
        Метод расчета общей Ж. 
        Логика: сумма состояний узлов, где UNKNOWN требует внимания.
        """
        total = sum(node.state.value for node in self.nodes)
        return total / len(self.nodes) if self.nodes else 0

    def find_uncertain_nodes(self) -> List[ArchitecturalNode]:
        """Авто-подсветка 'архитектурных дыр' (Блок C)"""
        return [node for node in self.nodes if node.state == State.UNKNOWN]

# Пример использования:
if __name__ == "__main__":
    n1 = ArchitecturalNode("Cut_Resistance", State.TRUE)
    n2 = ArchitecturalNode("Spark_Resistance", State.FALSE)
    n3 = ArchitecturalNode("Impact_Resistance", State.UNKNOWN) # Требует R&D

    engine = ViabilityEngine([n1, n2, n3])
    
    print(f"Общая жизнеспособность: {engine.calculate_system_viability():.2f}")
    print(f"Узлы под вопросом: {engine.find_uncertain_nodes()}")
